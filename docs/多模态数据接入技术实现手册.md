```

    ↓ ETL多模态处理
处理层（SQL→文本 / 文档分块 / Whisper转写）

    ↓ API调用

```

## STEP 1：结构化数据表 → 文本化

### 原理
MySQL 表格数据（车型参数、报价单、维修工单）无法直接向量化。
需要将每行数据模板化为"自然语言描述段落"再入向量库。

### 实现代码（Python）

```python
import pymysql

def export_car_params_to_docs():
    """将车型参数表每行转成一段自然语言文本"""
    conn = pymysql.connect(
        host='your-mysql-host', db='heguan_car',
        user='root', password='your-password',
        charset='utf8mb4'
    )
    cursor = conn.cursor(pymysql.cursors.DictCursor)
    cursor.execute("SELECT * FROM car_model_params")
    rows = cursor.fetchall()

    docs = []
    for row in rows:
        text = (
            f"车型：{row['model_name']}，品牌：{row['brand']}，"
            f"发动机：{row['engine']}，变速箱：{row['gearbox']}，"
            f"指导价：{row['price']}万元，质保政策：{row['warranty']}。"
            f"主要卖点：{row['selling_points']}。"
            f"适用人群：{row['target_user']}。"
        )
        docs.append({
            "content": text,
            "metadata": {
                "source_type": "structured_db",
                "table": "car_model_params",
                "row_id": row['id'],
                "model": row['model_name'],
                "brand": row['brand']
            }
        })
    conn.close()
    return docs


def export_repair_orders_to_docs():
    """将历史维修工单转成经验知识片段"""
    conn = pymysql.connect(
        host='your-mysql-host', db='heguan_car',
        user='root', password='your-password', charset='utf8mb4'
    )
    cursor = conn.cursor(pymysql.cursors.DictCursor)
    # 只取有解决方案的工单
    cursor.execute("""
        SELECT * FROM repair_orders 
        WHERE solution IS NOT NULL AND solution != ''
        ORDER BY create_time DESC LIMIT 5000
    """)
    rows = cursor.fetchall()

    docs = []
    for row in rows:
        text = (
            f"故障车型：{row['car_model']}，"
            f"故障描述：{row['fault_description']}，"
            f"维修方案：{row['solution']}，"
            f"更换配件：{row.get('parts_replaced', '无')}，"
            f"技师经验总结：{row.get('tech_notes', '')}。"
        )
        docs.append({
            "content": text,
            "metadata": {
                "source_type": "repair_order",
                "order_id": row['id'],
                "car_model": row['car_model'],
                "fault_category": row.get('fault_category', '未分类'),
                "create_time": str(row['create_time'])
            }
        })
    conn.close()
    return docs
```

## STEP 2：视频 → 带时间戳文字（核心关键）

### 原理
视频不能直接向量化。需要：
1. Whisper 语音识别 → 带时间戳文字段落
2. 每段保留 `start_sec / end_sec`，这是后续"数字人回答时跳转到对应视频片段"的基础
3. 建议每 30 秒或 200 字切割一段，保持语义完整

### 实现代码（Python）

```python
import whisper
import os

# 加载模型（首次运行会自动下载，约 1.5GB）
# 中文场景推荐 medium 或 large-v3
model = whisper.load_model("medium")


def video_to_chunks(video_path: str, video_id: str, video_title: str) -> list:
    """
    将视频文件转成带时间戳的知识片段列表
    返回格式：[{content, metadata{source_type, jump_url, start_sec, end_sec}}]
    """
    print(f"正在处理视频：{video_title}")
    result = model.transcribe(
        video_path,
        language="zh",
        word_timestamps=True,
        initial_prompt="以下是汽车维修和销售相关的专业内容"  # 给模型上下文提示，提升识别准确率
    )

    chunks = []
    buffer_text = ""
    start_time = 0.0
    CHUNK_MAX_SEC = 30     # 每段最多30秒
    CHUNK_MAX_CHARS = 200  # 每段最多200字

    for seg in result["segments"]:
        buffer_text += seg["text"].strip()
        time_elapsed = seg["end"] - start_time
        
        if time_elapsed >= CHUNK_MAX_SEC or len(buffer_text) >= CHUNK_MAX_CHARS:
            if buffer_text.strip():
                chunks.append({
                    "content": buffer_text.strip(),
                    "metadata": {
                        "source_type": "video",
                        "video_id": video_id,
                        "video_title": video_title,
                        "video_path": video_path,
                        "start_sec": round(start_time, 1),
                        "end_sec": round(seg["end"], 1),
                        # 前端通过这个 URL 直接跳到视频对应位置
                        "jump_url": f"/video/play?id={video_id}&t={int(start_time)}",
                        "time_label": f"{int(start_time//60):02d}:{int(start_time%60):02d}"
                    }
                })
            buffer_text = ""
            start_time = seg["end"]

    # 处理最后一段
    if buffer_text.strip():
        chunks.append({
            "content": buffer_text.strip(),
            "metadata": {
                "source_type": "video",
                "video_id": video_id,
                "video_title": video_title,
                "jump_url": f"/video/play?id={video_id}&t={int(start_time)}",
                "time_label": f"{int(start_time//60):02d}:{int(start_time%60):02d}",
                "start_sec": round(start_time, 1),
                "end_sec": round(result["segments"][-1]["end"], 1)
            }
        })

    print(f"  完成，共切割 {len(chunks)} 段")
    return chunks


def batch_process_videos(video_dir: str) -> list:
    """批量处理目录下所有视频"""
    all_chunks = []
    supported = ('.mp4', '.mov', '.avi', '.mkv')
    
    for filename in os.listdir(video_dir):
        if filename.lower().endswith(supported):
            video_id = os.path.splitext(filename)[0]
            video_path = os.path.join(video_dir, filename)
            chunks = video_to_chunks(video_path, video_id, video_id)
            all_chunks.extend(chunks)
    
    return all_chunks
```

## STEP 3：音频（接待录音/话术录音）→ 文字

```python
def audio_to_chunks(audio_path: str, audio_id: str, audio_title: str) -> list:
    """处理音频文件（逻辑同视频，去掉视频专属字段）"""
    result = model.transcribe(
        audio_path,
        language="zh",
        initial_prompt="以下是汽车销售接待和售后客服的对话内容"
    )

    chunks = []
    buffer_text = ""
    start_time = 0.0

    for seg in result["segments"]:
        buffer_text += seg["text"].strip()
        if seg["end"] - start_time >= 30 or len(buffer_text) >= 200:
            if buffer_text.strip():
                chunks.append({
                    "content": buffer_text.strip(),
                    "metadata": {
                        "source_type": "audio",
                        "audio_id": audio_id,
                        "audio_title": audio_title,
                        "start_sec": round(start_time, 1),
                        "end_sec": round(seg["end"], 1),
                        "jump_url": f"/audio/play?id={audio_id}&t={int(start_time)}"
                    }
                })
            buffer_text = ""
            start_time = seg["end"]

    return chunks
```

## STEP 4A：统一入库（Dify 低代码方案 · 推荐起步）

### 说明
Dify 已内置向量化+检索，只需将处理好的文本通过 API 上传即可。
Dify 会自动完成 Embedding → 向量存储 → 检索接口的全部工作。

```python
import requests
import json

DIFY_BASE_URL = "https://api.dify.ai/v1"
DIFY_API_KEY = "dataset-your-key-here"   # Dify 知识库 API Key
DATASET_ID = "your-dataset-id"           # 对应知识库 ID

def upload_docs_to_dify(docs: list, dataset_name: str):
    """批量上传处理好的文本片段到 Dify 知识库"""
    headers = {
        "Authorization": f"Bearer {DIFY_API_KEY}",
        "Content-Type": "application/json"
    }
    
    success = 0
    failed = 0
    
    for i, doc in enumerate(docs):
        # 将 metadata 拼接到文本末尾，方便检索时识别来源
        meta_suffix = f"\n\n[来源: {doc['metadata'].get('source_type', '未知')}]"
        if doc['metadata'].get('jump_url'):
            meta_suffix += f" [视频链接: {doc['metadata']['jump_url']}]"
        
        payload = {
            "name": f"{dataset_name}_{i:04d}",
            "text": doc["content"] + meta_suffix,
            "indexing_technique": "high_quality",
            "process_rule": {
                "rules": {
                    "pre_processing_rules": [
                        {"id": "remove_extra_spaces", "enabled": True},
                        {"id": "remove_urls_emails", "enabled": False}
                    ],
                    "segmentation": {
                        "separator": "\n",
                        "max_tokens": 500
                    }
                },
                "mode": "custom"
            }
        }
        
        resp = requests.post(
            f"{DIFY_BASE_URL}/datasets/{DATASET_ID}/document/create_by_text",
            headers=headers,
            json=payload
        )
        
        if resp.status_code == 200:
            success += 1
        else:
            failed += 1
            print(f"上传失败: {resp.text}")
        
        if i % 50 == 0:
            print(f"进度：{i}/{len(docs)}, 成功 {success}, 失败 {failed}")
    
    print(f"\n全部完成：成功 {success} 条，失败 {failed} 条")
```

## STEP 4B：私有化 Milvus 方案（高并发/数据安全）

```python
from pymilvus import (
    connections, FieldSchema, CollectionSchema,
    DataType, Collection, utility
)
from sentence_transformers import SentenceTransformer

# 连接 Milvus
connections.connect(host='localhost', port='19530')

# 加载中文向量模型（bge-large-zh 在汽车行业中文语义理解最佳）
embedder = SentenceTransformer('BAAI/bge-large-zh-v1.5')

COLLECTION_NAME = "heguan_knowledge_base"
VECTOR_DIM = 1024  # bge-large-zh 的维度

def create_collection():
    """创建知识库 Collection（含元数据字段）"""
    if utility.has_collection(COLLECTION_NAME):
        return Collection(COLLECTION_NAME)
    
    fields = [
        FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
        FieldSchema(name="content", dtype=DataType.VARCHAR, max_length=2000),
        FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=VECTOR_DIM),
        FieldSchema(name="source_type", dtype=DataType.VARCHAR, max_length=50),
        FieldSchema(name="jump_url", dtype=DataType.VARCHAR, max_length=500),
        FieldSchema(name="start_sec", dtype=DataType.FLOAT),
        FieldSchema(name="car_model", dtype=DataType.VARCHAR, max_length=100),
    ]
    
    schema = CollectionSchema(fields=fields, description="合群汽车知识库")
    collection = Collection(name=COLLECTION_NAME, schema=schema)
    
    # 创建 IVF_FLAT 索引（适合百万级数据）
    index_params = {"metric_type": "IP", "index_type": "IVF_FLAT", "params": {"nlist": 1024}}
    collection.create_index(field_name="embedding", index_params=index_params)
    
    return collection


def insert_docs_to_milvus(docs: list):
    """批量向量化并插入 Milvus"""
    collection = create_collection()
    
    BATCH_SIZE = 100  # 每批100条，防止内存溢出
    
    for start in range(0, len(docs), BATCH_SIZE):
        batch = docs[start:start+BATCH_SIZE]
        texts = [d["content"] for d in batch]
        
        # 向量化
        vectors = embedder.encode(
            texts,
            normalize_embeddings=True,
            show_progress_bar=False
        )
        
        # 插入数据
        data = [
            texts,
            vectors.tolist(),
            [d["metadata"].get("source_type", "") for d in batch],
            [d["metadata"].get("jump_url", "") for d in batch],
            [float(d["metadata"].get("start_sec", 0)) for d in batch],
            [d["metadata"].get("car_model", d["metadata"].get("model", "")) for d in batch],
        ]
        
        collection.insert(data)
        print(f"已插入 {min(start+BATCH_SIZE, len(docs))}/{len(docs)} 条")
    
    collection.flush()
    print("全部入库完成")
```

## STEP 5：RAG 检索 API（数字人调用核心）

```python
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
import openai  # 或替换为 vLLM 本地接口

app = FastAPI()

class QueryRequest(BaseModel):
    question: str
    role: str = "通用"  # 维修技师 / 销售顾问 / 售后客服

SYSTEM_PROMPT_TEMPLATE = """
你是合群汽车集团的专属AI导师"阿能"，专门负责24小时辅导员工。
当前提问者岗位：{role}

回答规则：
1. 只根据下方【知识库内容】回答，不要编造
2. 知识库没有的内容，直接说"这个问题我需要请教更有经验的老师傅，建议您联系技术支持"
3. 回答简洁实用，维修类问题给出步骤，销售类问题给出话术
4. 如果来源是视频，在回答末尾提示"可以查看原始视频演示"

【知识库内容】
{context}
"""

@app.post("/rag/query")
async def rag_query(req: QueryRequest):
    collection = Collection(COLLECTION_NAME)
    collection.load()
    
    # 1. 向量化问题
    q_vec = embedder.encode([req.question], normalize_embeddings=True)[0]
    
    # 2. 向量检索（Top-8）
    search_params = {"metric_type": "IP", "params": {"nprobe": 10}}
    results = collection.search(
        data=[q_vec.tolist()],
        anns_field="embedding",
        param=search_params,
        limit=8,
        output_fields=["content", "source_type", "jump_url", "start_sec", "car_model"]
    )
    
    # 3. 组装上下文 + 引用
    context_parts = []
    references = []
    
    for hit in results[0]:
        entity = hit.entity
        context_parts.append(entity.get("content", ""))
        
        if entity.get("source_type") == "video" and entity.get("jump_url"):
            references.append({
                "type": "video",
                "url": entity.get("jump_url"),
                "time_sec": entity.get("start_sec", 0),
                "time_label": f"{int(entity.get('start_sec',0)//60):02d}:{int(entity.get('start_sec',0)%60):02d}"
            })
    
    context = "\n\n---\n\n".join(context_parts)
    
    # 4. 调用 LLM（本地 Qwen2.5 / 远端 API 均可）
    messages = [
        {
            "role": "system",
            "content": SYSTEM_PROMPT_TEMPLATE.format(
                role=req.role,
                context=context
            )
        },
        {"role": "user", "content": req.question}
    ]
    
    # 调用本地 Qwen（vLLM 部署）
    response = openai.ChatCompletion.create(
        model="qwen2.5-7b-instruct",
        messages=messages,
        base_url="http://localhost:8000/v1",
        api_key="not-needed",
        stream=True
    )
    
    # 5. 返回答案 + 引用
    return {
        "answer": response.choices[0].message.content,
        "references": references
    }
```

## 关键注意事项

1. **视频时间戳一定要保留** — 这是数字人"说完答案、自动弹出视频原片"的基础，丢了就白处理
2. **结构化数据要用模板描述，不要直接存 JSON** — `"price: 18.5"` 不如 `"指导价18.5万元"` 检索命中率高
3. **音频/视频处理是离线批处理** — 新视频上传后触发异步任务，不要实时处理
4. **Whisper 建议用 medium 或 large-v3** — small 模型在汽车专业术语上识别率不足
5. **元数据必须随向量一起存** — 来源类型、时间戳、视频ID缺一不可，否则无法溯源
